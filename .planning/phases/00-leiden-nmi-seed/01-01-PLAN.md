       ```go
       func (d *leidenDetector) Detect(g *Graph) (CommunityResult, error) {
           // Guard clauses (directed, empty, single-node, zero-weight) — keep as-is
           ...

           // Seed!=0 → single run, deterministic
           if d.opts.Seed != 0 {
               return d.runOnce(g, d.opts.Seed)
           }

           // Resolve NumRuns: 0 → default 3
           effectiveNumRuns := d.opts.NumRuns
           if effectiveNumRuns == 0 {
               effectiveNumRuns = 3
           }
           // NumRuns=1 → single run with random seed
           if effectiveNumRuns == 1 {
               return d.runOnce(g, 0) // seed=0 → random in reset()
           }

           // Multi-run: each run gets baseSeed + i
           baseSeed := time.Now().UnixNano()
           var bestResult CommunityResult
           bestQ := math.Inf(-1)
           var lastErr error
           for i := 0; i < effectiveNumRuns; i++ {
               res, err := d.runOnce(g, baseSeed+int64(i))
               if err != nil {
                   lastErr = err
                   continue
               }
               if res.Modularity > bestQ {
                   bestQ = res.Modularity
                   bestResult = res
               }
           }
           if bestQ == math.Inf(-1) {
               return CommunityResult{}, lastErr
           }
           return bestResult, nil
       }
       ```

    4. Add `"time"` to the import block in leiden.go (already has `"math"`).

    - Do NOT check NumRuns before Seed. The `if d.opts.Seed != 0` check MUST come first.
    - Each runOnce call MUST independently acquireLeidenState/releaseLeidenState — do NOT share pool state across runs.
    - baseSeed MUST be computed ONCE before the loop, NOT inside each iteration.
    - bestResult stores the full CommunityResult (Passes/Moves from the best-Q run only, NOT summed).
    - If all runs error, return lastErr. If any run succeeds, return bestResult even if other runs errored.
