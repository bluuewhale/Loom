       // computeAffected returns the set of nodes whose ego-nets must be recomputed
       // for the given delta. Affected = new nodes + neighbors of all edge endpoints
       // (queried on the already-updated graph g).
       func computeAffected(g *Graph, delta GraphDelta) map[NodeID]struct{} {
           affected := make(map[NodeID]struct{})
           for _, n := range delta.AddedNodes {
               affected[n] = struct{}{}
           }
           for _, e := range delta.AddedEdges {
               affected[e.From] = struct{}{}
               affected[e.To] = struct{}{}
               for _, nb := range g.Neighbors(e.From) {
                   affected[nb.To] = struct{}{}
               }
               for _, nb := range g.Neighbors(e.To) {
                   affected[nb.To] = struct{}{}
               }
           }
           return affected
       }
       ```

       func buildPersonaGraphIncremental(
           g *Graph,
           affected map[NodeID]struct{},
           prior OverlappingCommunityResult,
           localDetector CommunityDetector,
       ) (*Graph, map[NodeID]map[int]NodeID, map[NodeID]NodeID, map[NodeID]map[NodeID]int, map[NodeID]int, error)

       Returns: personaGraph, personaOf, inverseMap, partitions, warmPartition, error.

       Implementation steps:
       a. Deep-copy prior.personaOf, prior.inverseMap, prior.partitions into new maps (shallow copy of inner maps is fine for unaffected nodes since we only overwrite affected entries).
       b. Compute nextPersona = max(max over all keys in prior.inverseMap, max over g.Nodes()) + 1. This satisfies ONLINE-11.
       c. For each node v in affected:
          - Delete old persona entries: for each (commID, personaID) in personaOf[v], delete inverseMap[personaID]. Then delete personaOf[v] and partitions[v].
       d. For each node v in affected:
          - Build ego-net: egoNet := buildEgoNet(g, v)
          - If egoNet.NodeCount() == 0: assign single persona (community 0), update personaOf[v], inverseMap, partitions[v] = empty map
          - Else: run localDetector.Detect(egoNet), store partitions[v] = result.Partition, assign persona per unique community using nextPersona++
       e. Build warmPartition: copy prior.personaPartition, but omit any PersonaID not in the new inverseMap (deleted affected personas). New personas are absent = Louvain assigns singleton automatically.
       f. Rebuild persona graph from scratch (Graph has no RemoveNode):
          - Create NewGraph(false)
          - AddNode for every PersonaID in inverseMap
          - Wire edges using the SAME canonical [2]NodeID{lo,hi} dedup pattern from buildPersonaGraph. Iterate ALL edges in g (for u in g.Nodes(), for e in g.Neighbors(u)), compute commOfVinGu from partitions[u][v], commOfUinGv from partitions[v][u], look up personaOf[u][commOfVinGu] and personaOf[v][commOfUinGv], add edge if both exist.
          NOTE: This rebuilds the full persona graph edges, which is correct — the partitions map is already patched so only affected nodes have new partition assignments. The cost is O(|E|) where E is edges in the original graph, but this is unavoidable without RemoveNode. Phase 12 will parallelize if needed.

       func (d *egoSplittingDetector) Update(g *Graph, delta GraphDelta, prior OverlappingCommunityResult) (OverlappingCommunityResult, error) {
           if g.IsDirected() {
               return OverlappingCommunityResult{}, ErrDirectedNotSupported
           }
           if len(delta.AddedNodes) == 0 && len(delta.AddedEdges) == 0 {
               return prior, nil
           }
           // Graceful fallback if prior lacks carry-forward fields
           if prior.personaOf == nil || prior.inverseMap == nil || prior.partitions == nil {
               return d.Detect(g)
           }

           // Step 1: Compute affected node set (ONLINE-05)
           affected := computeAffected(g, delta)

           // Step 2: Incremental persona graph build (ONLINE-06, ONLINE-11)
           personaGraph, newPersonaOf, newInverseMap, newPartitions, warmPartition, err :=
               buildPersonaGraphIncremental(g, affected, prior, d.opts.LocalDetector)
           if err != nil {
               return OverlappingCommunityResult{}, err
           }

           // Step 3: Warm-start global detection (ONLINE-07)
           warmGlobal := warmStartedDetector(d.opts.GlobalDetector, warmPartition)
           globalResult, err := warmGlobal.Detect(personaGraph)
           if err != nil {
               return OverlappingCommunityResult{}, err
           }

           // Step 4: Map personas back to original nodes (same as Detect steps 3-5)
           nodeCommunities := mapPersonasToOriginal(globalResult.Partition, newInverseMap)

           // Step 4b: Deduplicate community IDs per node (same as Detect)
           for node, comms := range nodeCommunities {
               seen := make(map[int]struct{}, len(comms))
               unique := comms[:0]
               for _, c := range comms {
                   if _, ok := seen[c]; !ok {
                       seen[c] = struct{}{}
                       unique = append(unique, c)
                   }
               }
               nodeCommunities[node] = unique
           }

           // Step 5: Build Communities [][]NodeID (same as Detect)
           maxComm := -1
           for _, comms := range nodeCommunities {
               for _, c := range comms {
                   if c > maxComm {
                       maxComm = c
                   }
               }
           }
           communities := make([][]NodeID, maxComm+1)
           for node, comms := range nodeCommunities {
               for _, c := range comms {
                   communities[c] = append(communities[c], node)
               }
           }
           var filtered [][]NodeID
           commRemap := make(map[int]int)
           for i, members := range communities {
               if len(members) > 0 {
                   commRemap[i] = len(filtered)
                   filtered = append(filtered, members)
               }
           }
           for node, comms := range nodeCommunities {
               remapped := make([]int, len(comms))
               for j, c := range comms {
                   remapped[j] = commRemap[c]
               }
               nodeCommunities[node] = remapped
           }

           return OverlappingCommunityResult{
               Communities:      filtered,
               NodeCommunities:  nodeCommunities,
               personaOf:        newPersonaOf,
               inverseMap:       newInverseMap,
               partitions:       newPartitions,
               personaPartition: globalResult.Partition,
           }, nil
       }
       ```
