**graph/louvain_state.go** -- Modify `reset()` signature and add warm-seed logic (per D-02, D-03, D-04, D-05):

Change signature from:
```go

```
to:
```go
func (st *louvainState) reset(g *Graph, seed int64, initialPartition map[NodeID]int)
```

Keep the existing clear/RNG/sort logic unchanged. Replace the singleton population loop
(lines 67-72: `for i, n := range nodes { st.partition[n] = i; st.commStr[i] = g.Strength(n) }`)
with:

```go
if initialPartition == nil {
    // Cold start: trivial singleton assignment (existing behavior)
    for i, n := range nodes {
        st.partition[n] = i
        st.commStr[i] = g.Strength(n)
    }
    return
}

// Warm start: seed from initialPartition (D-02)
// Step 1: find max community ID for new-node singleton offset (D-03)
maxCommID := -1
for _, c := range initialPartition {
    if c > maxCommID {
        maxCommID = c
    }
}
nextNewComm := maxCommID + 1

// Step 2: assign partition, new nodes get fresh singletons (D-03, D-04)
for _, n := range nodes {
    if c, ok := initialPartition[n]; ok {
        st.partition[n] = c
    } else {
        st.partition[n] = nextNewComm
        nextNewComm++
    }
}

// Step 3: compact to 0-indexed contiguous IDs (D-05)
remap := make(map[int]int, len(nodes))
next := 0
for _, n := range nodes { // nodes already sorted -- deterministic remap
    c := st.partition[n]
    if _, exists := remap[c]; !exists {
        remap[c] = next
        next++
    }
    st.partition[n] = remap[c]
}

// Step 4: build commStr from CURRENT graph strengths (CRITICAL -- not from prior run)
for _, n := range nodes {
    st.commStr[st.partition[n]] += g.Strength(n)
}
```
