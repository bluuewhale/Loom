```go
// partitionToGroundTruth converts a map[int]int partition (nodeID -> commID)
// into [][]NodeID (community-first format) suitable for OmegaIndex.
func partitionToGroundTruth(partition map[int]int) [][]NodeID {
    commMap := make(map[int][]NodeID)
    for node, comm := range partition {
        commMap[comm] = append(commMap[comm], NodeID(node))
    }
    // Collect communities in sorted order by community ID for determinism
    maxComm := -1
    for c := range commMap {
        if c > maxComm {
            maxComm = c
        }
    }
    result := make([][]NodeID, 0, len(commMap))
    for i := 0; i <= maxComm; i++ {
        if members, ok := commMap[i]; ok {
            result = append(result, members)
        }
    }
    return result
}
```

```go
func TestEgoSplittingConcurrentDetect(t *testing.T) {
    // 4 goroutines each running Detect on distinct graph instances
    graphs := make([]*Graph, 4)
    for i := range graphs {
        graphs[i] = buildGraph(testdata.KarateClubEdges)
    }

    var wg sync.WaitGroup
    for i, g := range graphs {
        wg.Add(1)
        go func(g *Graph, idx int) {
            defer wg.Done()
            det := NewEgoSplitting(EgoSplittingOptions{
                LocalDetector:  NewLouvain(LouvainOptions{Seed: int64(idx + 1)}),
                GlobalDetector: NewLouvain(LouvainOptions{Seed: int64(idx + 1)}),
            })
            for j := 0; j < 5; j++ {
                result, err := det.Detect(g)
                if err != nil {
                    t.Errorf("goroutine %d iteration %d: %v", idx, j, err)
                    return
                }
                if len(result.Communities) == 0 {
                    t.Errorf("goroutine %d iteration %d: no communities", idx, j)
                    return
                }
            }
        }(g, i)
    }
    wg.Wait()
}
```

This test validates EGO-10 when run with `go test -race`. The design is inherently race-safe (state-per-call in Detect, buildPersonaGraph, mapPersonasToOriginal all allocate fresh maps), but this test confirms it under the race detector.

```go
// BenchmarkEgoSplitting10K measures EgoSplitting on a 10K-node Barabasi-Albert graph.
// Target: <= 300ms/op. Uses the shared bench10K graph (10K nodes, ~50K edges, BA model).
func BenchmarkEgoSplitting10K(b *testing.B) {
    det := NewEgoSplitting(EgoSplittingOptions{
        LocalDetector:  NewLouvain(LouvainOptions{Seed: 1}),
        GlobalDetector: NewLouvain(LouvainOptions{Seed: 1}),
    })
    // Warmup: one full run to populate sync.Pool in underlying Louvain detectors
    det.Detect(bench10K)
    b.ResetTimer()
    b.ReportAllocs()
    for i := 0; i < b.N; i++ {
        det.Detect(bench10K)
    }
}
```

**Key details:**
- Reuses the existing `bench10K` variable (10K-node BA graph initialized in `init()`) -- no new graph generation needed.
- Seeded detectors for determinism (Seed: 1, matching existing benchmark pattern).
- Warmup call before b.ResetTimer() to populate sync.Pool (same pattern as BenchmarkLouvain10K).
- The OverlappingCommunityDetector interface uses `Detect(*Graph)` not the concrete type, so `det.Detect(bench10K)` compiles.
